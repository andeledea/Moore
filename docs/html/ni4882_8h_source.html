208// Address type (for 488.2 calls)

210typedef short Addr4882_t; // System dependent: must be 16 bits

212//
213// This macro can be used to easily create an entry in address list
214// that is required by many of the 488.2 functions. The primary address goes in the
215// lower 8-bits and the secondary address goes in the upper 8-bits.
216//
217#define MakeAddr(pad, sad) ((Addr4882_t)(((pad) & 0xFF) | ((sad) << 8)))
218
219//
220// This value is used to terminate an address list. It should be
221// assigned to the last entry.
222//
223#ifndef NOADDR
224#define NOADDR ((Addr4882_t)(-1))
225#endif
226
227//
228// The following two macros are used to "break apart" an address list
229// entry. They take an unsigned integer and return either the primary
230// or secondary address stored in the integer.
231//
232#define GetPAD(val) ((val) & 0xFF)
233#define GetSAD(val) (((val) >> 8) & 0xFF)
